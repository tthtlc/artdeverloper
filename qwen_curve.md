def generate_plot(curve):
    """Generates a plot for a given curve dictionary and returns base64 string."""
    t_min, t_max = curve['range']
    t = np.linspace(t_min, t_max, 1000)
    
    try:
        x, y = curve['eq'](t)
        
        # Handle potential infinities/nans for display
        mask = np.isfinite(x) & np.isfinite(y)
        x, y = x[mask], y[mask]
        
        plt.figure(figsize=(4, 4))
        plt.plot(x, y, 'b-', linewidth=1.5)
        plt.title(f"{curve['name']}", fontsize=10)
        plt.xlabel(f"Math: {curve['math']}", fontsize=8)
        plt.grid(True, linestyle='--', alpha=0.5)
        plt.axis('equal')
        plt.xticks([])
        plt.yticks([])
        
        # Save to buffer
        buf = BytesIO()
        plt.savefig(buf, format='png', bbox_inches='tight')
        plt.close()
        buf.seek(0)
        img_str = base64.b64encode(buf.read()).decode('utf-8')
        return img_str
    except Exception as e:
        print(f"Error plotting {curve['name']}: {e}")
        return None
